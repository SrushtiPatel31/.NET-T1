using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace P17
{
    public delegate void AlarmHandler(String loc);

    class AlarmSystem
    {
        public event AlarmHandler OnAlarmTriggered;
        public void TriggerAlarm(String loc)
        {
            Console.WriteLine("Alarm Triggered");
            OnAlarmTriggered.Invoke(loc);
        }
    }
    internal class Program
    {
        static void Main(string[] args)
        {
            AlarmSystem a = new AlarmSystem();

            a.OnAlarmTriggered += loc => Console.WriteLine($"Security called for {loc}");

            a.OnAlarmTriggered += loc => Console.WriteLine($"Log saved for {loc}");

            a.OnAlarmTriggered += loc => Console.WriteLine($"Sprinkles activated for {loc}");

            a.TriggerAlarm("main Gate");
        }
    }
}
